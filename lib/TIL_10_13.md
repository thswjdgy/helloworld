void main() {
  var size = 10;
  for (var i = 0; i < size; i++) {
    printSharp(size, i);
  }
}

void printSharp(int size, int y) {
  var result = "";
  for (var x = 0; x < size; x++) {
    var condition = 
      x == y ||
      size - 1 - x == y;
    if (condition) {
       result += "#";
    }
    else {
      result += '.';  
    }
  }
  print(result);
}

void printSharp(int size) {
  for (int i = 0; i < size; i++) {
    var x = '';
    for (int y = 0; y < size; y++) {
      if (i == 0 || i == size - 1) {
        x += '#'; 
      } else if (y == 0 || y == size - 1) {
        x += '#'; 
      } else if (y == i || y == size - i - 1) {
        x += '#'; 
      } else {
        x += ' ';
      }
    }
    print(x);
  }
}

void main() {
  printSharp(20);
}

  var n = 4;

    var line = "";
    var width = n * 2 - 1;
    
    for (var x = 0; x < width * n; x++) {
      var condition =
        (x % width) - (n - 1) == y ||
        n - (x % width) - 1 == y;
         
      if(condition) {
      line += "#";
    }

      line += " ";
    }
  }
